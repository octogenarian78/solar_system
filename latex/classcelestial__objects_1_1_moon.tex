\doxysubsection{Description détaillée}
\begin{DoxyVerb}@class Moon
@brief Classe représentant une lune.

Hérite de la classe CelestialBody et ajoute des informations sur l'orbite autour d'une planète.

@param name: Nom de la lune.
@param size: Taille de la lune.
@param planet: Planète autour de laquelle la lune orbite.
@param orbit_radius: Rayon de l'orbite en Unités Astronomiques.
@param orbit_speed: Vitesse orbitale de la lune.
\end{DoxyVerb}
